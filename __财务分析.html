def calculate_income(base, rate, rebate):
    bonus = 0
    if rate >= 0.9:

    rebate_bonus = 35 * 15 * 18 * rebate  # 示例计算

results = [calculate_income(b, r, rb) for b, r, rb in zip(base_salary, completion_rate, rebate_ratio)]
